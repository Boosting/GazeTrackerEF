-- включение гита. после этой команды каталог где она выполнена становится 
   под версионный контроль и можно выполнять все гитовые команды
git init

-- если нужно закачать проги с удаленного сервера то нужно выполнить
git clone  https://github.com/danmas/GazeTrackerEF.git 

-- если создал или скопировал файл то его нужно добавить в контроль версий (* - занчит все)
git add *

-- посмотреть что творится в каталоге можно
   увидишь в какой ветке сидишь, что модифицировано, добавлено, удалено
git status

-- чтобы закоммитить последние изменения нужно выполнить
git commit -a -m "Добавил Dasher в AppForEF"

-- если ветка подключена к удаленному серверу, то сбросить изменения на сервер можно командой git push
   выполняется только после коммита
git push

-- посмотреть имеющиеся ветки
git branch

-- создать ветку develop и сразу перейти в неё
git checkout -b  develop

-- перейти на ветку master
git checkout master

-- слить изменения текущей ветки с веткой develop
git merge develop

-- подключить ветку develop к удаленному серверу, после этого, находясь в ней можно делать git push 
-- и изменения из неё накатятся на сервер 
git push --set-upstream origin develop
